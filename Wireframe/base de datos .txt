-- =====================================================
-- TABLAS DE AUTENTICACIÓN (módulo AUTH_)
-- Prefijo de atributo: user_ / group_ / permission_
-- Las llaves primarias y foráneas mantienen el patrón
-- id_entidad ya que así quedan ya identificadas sin
-- necesidad de repetir el prefijo (evita redundancia
-- tipo id_user_user_id).
-- =====================================================

CREATE TABLE AUTH_USER (
    id_user INT AUTO_INCREMENT PRIMARY KEY,
    user_username VARCHAR(150) NOT NULL UNIQUE,
    user_email VARCHAR(255) NOT NULL UNIQUE,
    user_password VARCHAR(255) NOT NULL,
    user_first_name VARCHAR(150),
    user_last_name VARCHAR(150),
    user_dni VARCHAR(20) UNIQUE,
    user_phone VARCHAR(20) UNIQUE,
    user_profile_photo VARCHAR(500),
    user_is_active BOOLEAN DEFAULT TRUE,
    user_is_superuser BOOLEAN DEFAULT FALSE
);

CREATE TABLE AUTH_GROUP (
    id_group INT AUTO_INCREMENT PRIMARY KEY,
    group_name VARCHAR(150) NOT NULL
);

CREATE TABLE AUTH_PERMISSION (
    id_permission INT AUTO_INCREMENT PRIMARY KEY,
    permission_name VARCHAR(150) NOT NULL,
    permission_codename VARCHAR(150) NOT NULL,
    permission_content_type INT
);

CREATE TABLE AUTH_USER_GROUPS (
    id_user_group INT AUTO_INCREMENT PRIMARY KEY,
    id_user INT NOT NULL,
    id_group INT NOT NULL,

    CONSTRAINT fk_usergroups_user
        FOREIGN KEY (id_user)
        REFERENCES AUTH_USER(id_user),

    CONSTRAINT fk_usergroups_group
        FOREIGN KEY (id_group)
        REFERENCES AUTH_GROUP(id_group)
);

CREATE TABLE AUTH_GROUP_PERMISSIONS (
    id_group_permission INT AUTO_INCREMENT PRIMARY KEY,
    id_group INT NOT NULL,
    id_permission INT NOT NULL,

    CONSTRAINT fk_groupperm_group
        FOREIGN KEY (id_group)
        REFERENCES AUTH_GROUP(id_group),

    CONSTRAINT fk_groupperm_permission
        FOREIGN KEY (id_permission)
        REFERENCES AUTH_PERMISSION(id_permission)
);

CREATE TABLE AUTH_USER_PERMISSION (
    id_user_permission INT AUTO_INCREMENT PRIMARY KEY,
    id_user INT NOT NULL,
    id_permission INT NOT NULL,

    CONSTRAINT fk_userperm_user
        FOREIGN KEY (id_user)
        REFERENCES AUTH_USER(id_user),

    CONSTRAINT fk_userperm_permission
        FOREIGN KEY (id_permission)
        REFERENCES AUTH_PERMISSION(id_permission)
);

-- =====================================================
-- MÓDULO AGRÍCOLA (AGRO_)
-- Prefijo de atributo: farm_ / plot_ / context_ /
-- conversation_ / message_ / analysis_ / history_
-- =====================================================

CREATE TABLE AGRO_FARM (
    id_farm INT AUTO_INCREMENT PRIMARY KEY,
    id_user INT NOT NULL,
    farm_name VARCHAR(255) NOT NULL,
    farm_location TEXT,

    CONSTRAINT fk_farm_user
        FOREIGN KEY (id_user)
        REFERENCES AUTH_USER(id_user)
);

CREATE TABLE AGRO_PLOT (
    id_plot INT AUTO_INCREMENT PRIMARY KEY,
    id_farm INT NOT NULL,
    plot_name VARCHAR(255) NOT NULL,
    plot_hectares DECIMAL(10,2),
    plot_gps_location VARCHAR(255),
    plot_zone VARCHAR(100),
    plot_rows INT,

    CONSTRAINT fk_plot_farm
        FOREIGN KEY (id_farm)
        REFERENCES AGRO_FARM(id_farm)
);

CREATE TABLE AGRO_CONTEXT (
    id_context INT AUTO_INCREMENT PRIMARY KEY,
    id_plot INT NOT NULL,
    context_plant_key_or_id VARCHAR(255),
    context_affected_part VARCHAR(255),
    context_main_symptom VARCHAR(255),
    context_status VARCHAR(100),
    context_created_at DATETIME DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_context_plot
        FOREIGN KEY (id_plot)
        REFERENCES AGRO_PLOT(id_plot)
);

CREATE TABLE AGRO_CONVERSATION (
    id_conversation INT AUTO_INCREMENT PRIMARY KEY,
    id_context INT NOT NULL,
    conversation_title VARCHAR(255),
    conversation_created_at DATETIME DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_conversation_context
        FOREIGN KEY (id_context)
        REFERENCES AGRO_CONTEXT(id_context)
);

CREATE TABLE AGRO_CHAT_MESSAGE (
    id_chat_message INT AUTO_INCREMENT PRIMARY KEY,
    id_conversation INT NOT NULL,
    message_role VARCHAR(50),
    message_content TEXT,
    message_image_type VARCHAR(100),
    message_image_path VARCHAR(500),
    message_created_at DATETIME DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_chat_conversation
        FOREIGN KEY (id_conversation)
        REFERENCES AGRO_CONVERSATION(id_conversation)
);

CREATE TABLE AGRO_ANALYSIS_RESULT (
    id_analysis_result INT AUTO_INCREMENT PRIMARY KEY,
    id_conversation INT NOT NULL,
    id_chat_message INT NOT NULL,
    analysis_image_path VARCHAR(500),
    analysis_disease_name_predicted VARCHAR(255),
    analysis_confidence FLOAT,
    analysis_probability FLOAT,
    analysis_severity VARCHAR(100),
    analysis_text TEXT,
    analysis_recommendations_text TEXT,
    analysis_created_at DATETIME DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_analysis_conversation
        FOREIGN KEY (id_conversation)
        REFERENCES AGRO_CONVERSATION(id_conversation),

    CONSTRAINT fk_analysis_message
        FOREIGN KEY (id_chat_message)
        REFERENCES AGRO_CHAT_MESSAGE(id_chat_message)
);

CREATE TABLE AGRO_PLANT_HISTORY (
    id_plant_history INT AUTO_INCREMENT PRIMARY KEY,
    id_context INT NOT NULL,
    id_analysis_result INT NOT NULL,
    history_final_diagnosis VARCHAR(255),
    history_treatment_applied TEXT,
    history_notes TEXT,
    history_updated_at DATETIME,
    history_created_at DATETIME DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_history_context
        FOREIGN KEY (id_context)
        REFERENCES AGRO_CONTEXT(id_context),

    CONSTRAINT fk_history_analysis
        FOREIGN KEY (id_analysis_result)
        REFERENCES AGRO_ANALYSIS_RESULT(id_analysis_result)
);

-- =====================================================
-- MÓDULO RAG (RAG_)
-- Prefijo de atributo: document_ / chunk_
-- =====================================================

CREATE TABLE RAG_DOCUMENT (
    id_document INT AUTO_INCREMENT PRIMARY KEY,
    document_title VARCHAR(255) NOT NULL,
    document_source_path VARCHAR(500) NOT NULL UNIQUE,
    document_file_hash VARCHAR(255),
    document_chunks_count INT,
    document_created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    document_updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
        ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE RAG_CHUNK (
    id_chunk INT AUTO_INCREMENT PRIMARY KEY,
    id_document INT NOT NULL,
    chunk_index INT NOT NULL,
    chunk_text TEXT,
    chunk_page INT,
    chunk_embedding LONGBLOB,
    chunk_embedding_dim INT,
    chunk_embedding_model VARCHAR(255),
    chunk_created_at DATETIME DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_chunk_document
        FOREIGN KEY (id_document)
        REFERENCES RAG_DOCUMENT(id_document),

    CONSTRAINT uk_chunk_document_index
        UNIQUE (id_document, chunk_index)
);
